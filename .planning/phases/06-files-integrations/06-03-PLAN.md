```typescript
export const runtime = 'nodejs'

import { renderToBuffer } from '@react-pdf/renderer'
import { getOCById } from '@/actions/oc'
import { OCPDFDocument } from '@/lib/pdf/OCPDFDocument'
import React from 'react'

export async function GET(
  _req: Request,
  { params }: { params: Promise<{ id: string }> }
) {
  const { id } = await params
  const result = await getOCById(id)
  if ('error' in result) {
    return new Response('OC no encontrada', { status: 404 })
  }

  const oc = result.data.oc
  const buffer = await renderToBuffer(React.createElement(OCPDFDocument, { oc }))

  return new Response(buffer, {
    headers: {
      'Content-Type': 'application/pdf',
      'Content-Disposition': `attachment; filename="OC-${oc.referenciaOC}.pdf"`,
    },
  })
}
```

Nota: Usar `React.createElement` en lugar de JSX en el Route Handler evita problemas con el transform JSX en archivos `.ts`. Si el archivo es `.tsx`, puede usarse JSX directamente: `renderToBuffer(<OCPDFDocument oc={oc} />)`. Crear como `.tsx`.

**`src/app/(importador)/importador/dashboard/page.tsx`**

```typescript
'use client'
import { useState } from 'react'
import { FileText, Truck, Package, CheckCircle2 } from 'lucide-react'
import { MOCK_OCS } from '@/lib/mock-ocs'
import type { OC, EstadoOC } from '@/lib/mock-ocs'
import { StatCard } from '@/components/dashboard/StatCard'
import { OCTable } from '@/components/dashboard/OCTable'
import { FilterBar } from '@/components/dashboard/FilterBar'

export default function ImportadorDashboardPage() {
  const [searchQuery, setSearchQuery] = useState('')
  const [estadoFiltro, setEstadoFiltro] = useState<EstadoOC | ''>('')

  // Importador ve TODAS las OCs
  const todasLasOCs = MOCK_OCS

  // Filtrado client-side
  const ocsFiltradas = todasLasOCs.filter((oc) => {
    const matchProveedor = oc.proveedor.toLowerCase().includes(searchQuery.toLowerCase())
    const matchEstado = estadoFiltro === '' || oc.estado === estadoFiltro
    return matchProveedor && matchEstado
  })

  // Stat cards calculadas desde todas las OCs del importador (no filtradas)
  const stats = {
    totales: todasLasOCs.length,
    enTransito: todasLasOCs.filter((oc) => oc.estado === 'en_transito').length,
    enAduana: todasLasOCs.filter((oc) => oc.estado === 'en_aduana').length,
    entregadas: todasLasOCs.filter((oc) => oc.estado === 'entregada').length,
  }

  const hasFilters = searchQuery !== '' || estadoFiltro !== ''

  return (
    <div className="p-6 bg-fondo min-h-screen">
      <h1 className="text-xl font-bold text-titulares mb-6">Dashboard</h1>
      <div className="grid grid-cols-2 lg:grid-cols-4 gap-4 mb-8">
        <StatCard icon={FileText} value={stats.totales} label="OC Totales" />
        <StatCard icon={Truck} value={stats.enTransito} label="En tránsito" />
        <StatCard icon={Package} value={stats.enAduana} label="En aduana" />
        <StatCard icon={CheckCircle2} value={stats.entregadas} label="Entregadas" />
      </div>
      <FilterBar onSearch={setSearchQuery} onEstado={setEstadoFiltro} />
      <OCTable
        ocs={ocsFiltradas}
        rol="importador"
      />
    </div>
  )
}
```

**Nota:** Cuando `ocsFiltradas.length === 0`, OCTable internamente renderiza `<EmptyState rol="importador" hasFilters={hasFilters} />`. La página no necesita manejar el empty state directamente.

## Trust Boundaries

| Boundary | Descripción |
|----------|-------------|
| Input de filtro (texto) | El usuario escribe texto que se usa en `.includes()` — no se persiste ni se envía al servidor |
| Rol hardcodeado en página | En Phase 2 el rol es string literal en el JSX — en Phase 5 vendrá de Clerk JWT |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-03-01 | Tampering | FilterBar input | accept | El texto del filtro solo afecta rendering client-side de datos mock. En Phase 5 los filtros serán parámetros de query Mongoose validados en Server Actions. |
| T-02-03-02 | Elevation of Privilege | Rol hardcodeado en pages | accept | En Phase 2 el rol es literal de string en el source code. En Phase 5, el rol proviene exclusivamente de `sessionClaims.metadata.role` del JWT de Clerk en un Server Component — no modificable por el cliente. |
| T-02-03-03 | Information Disclosure | Filtrado mock por email | accept | Los emails de filtrado son strings ficticios hardcodeados. En Phase 5, el filtro usará el email del usuario autenticado (`currentUser()` de Clerk) — nunca input del cliente. |
| T-02-03-04 | Spoofing | Delete action (Phase 2) | accept | `onConfirm` del modal no ejecuta eliminación real en Phase 2. En Phase 5, el Server Action `deleteOC` verificará que `sessionClaims.metadata.role === 'importador'` antes de ejecutar cualquier operación en MongoDB. |
</threat_model>

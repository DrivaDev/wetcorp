    ```ts
    import { clerkMiddleware, createRouteMatcher } from '@clerk/nextjs/server'
    import { NextResponse } from 'next/server'

    const isPublicRoute  = createRouteMatcher(['/sign-in(.*)', '/sign-up(.*)'])
    const isOnboarding   = createRouteMatcher(['/onboarding(.*)'])
    const isImportador   = createRouteMatcher(['/importador(.*)'])
    const isProveedor    = createRouteMatcher(['/proveedor(.*)'])
    const isDespachante  = createRouteMatcher(['/despachante(.*)'])

    export default clerkMiddleware(async (auth, req) => {
      const { isAuthenticated, sessionClaims, redirectToSignIn } = await auth()
      const role = sessionClaims?.metadata?.role

      if (isAuthenticated && isOnboarding(req)) return NextResponse.next()
      if (!isAuthenticated && !isPublicRoute(req)) return redirectToSignIn({ returnBackUrl: req.url })

      if (isAuthenticated && !role && !isOnboarding(req)) {
        return NextResponse.redirect(new URL('/onboarding', req.url))
      }

      if (isAuthenticated && role) {
        if (isImportador(req) && role !== 'importador') {
          return NextResponse.redirect(new URL('/importador/dashboard', req.url))
        }
        if (isProveedor(req) && role !== 'proveedor') {
          return NextResponse.redirect(new URL('/proveedor/dashboard', req.url))
        }
        if (isDespachante(req) && role !== 'despachante') {
          return NextResponse.redirect(new URL('/despachante/dashboard', req.url))
        }
      }
    })

    export const config = {
      matcher: [
        '/((?!_next|[^?]*\\.(?:html?|css|js(?!on)|jpe?g|webp|png|gif|svg|ttf|woff2?|ico|csv|docx?|xlsx?|zip|webmanifest)).*)',
        '/(api|trpc)(.*)',
        '/__clerk/(.*)',
      ],
    }
    ```

    SECURITY NOTE: The `afterSignOutUrl` in UserButton components (in Sidebar and Navbar, created in Plan 03) MUST be hardcoded as `"/sign-in"`. It must never be read from a query parameter — that would be an open redirect vulnerability.

       ```ts
       import { auth, clerkClient } from '@clerk/nextjs/server'

       type Role = 'importador' | 'proveedor' | 'despachante'

       export async function completeOnboarding(role: Role): Promise<{ success: true } | { error: string }> {
         const { userId } = await auth()
         if (!userId) return { error: 'No autenticado' }

         const client = await clerkClient()
         await client.users.updateUser(userId, {
           publicMetadata: { role },
         })
         return { success: true }
       }
       ```
       CRITICAL: `clerkClient` must be `await`-ed (Clerk v6+ breaking change). Using it without `await` returns a promise, not the client.

       ```tsx
       type Role = 'importador' | 'proveedor' | 'despachante'

       const ROLE_REDIRECTS: Record<Role, string> = {

         proveedor:   '/proveedor/dashboard',

               {ROLES.map(({ id, Icon, title, description }) => (
                 <button
                   key={id}
                   onClick={() => setSelected(id)}
                   className={cn(
                     'border-2 rounded-xl p-6 cursor-pointer text-left transition-colors duration-150',
                     'active:scale-[0.98] transition-transform duration-75',

                 <button
                   onClick={handleConfirm}
                   disabled={loading}
                   className="bg-principal text-white px-8 py-3 rounded-lg font-medium hover:bg-titulares transition-colors duration-150 min-h-[44px] disabled:opacity-60 disabled:cursor-not-allowed inline-flex items-center gap-2"
                 >
                   {loading && <Loader2 size={18} className="animate-spin" />}

               </div>
             )}

             {error && (
               <p className="text-center text-red-600 text-sm mt-4">{error}</p>
             )}
           </div>
         )
       ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Client → Server Action (onboarding) | Role value comes from client button click — must be validated server-side |
| JWT → Middleware | Role in JWT must be written server-side only (publicMetadata) |
| Sign-out redirect → Browser | afterSignOutUrl must be hardcoded, never from query param |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-01 | Tampering | `completeOnboarding` Server Action | mitigate | Role parameter typed as `Role = 'importador' \| 'proveedor' \| 'despachante'` — TypeScript rejects invalid values; `publicMetadata` is server-write only per Clerk architecture |
| T-02-02 | Elevation of Privilege | Middleware route protection | mitigate | `clerkMiddleware` runs at the edge on every request; wrong-role users redirected to their own dashboard; unauthenticated users redirect to sign-in |
| T-02-03 | Spoofing | Open redirect on sign-out | mitigate | `afterSignOutUrl="/sign-in"` hardcoded string in all UserButton usages — never `router.query.redirect` or similar |
| T-02-04 | Elevation of Privilege | Stale JWT after role assignment | mitigate | `await user?.reload()` called in onboarding page before redirect — forces JWT re-issuance with new publicMetadata |
| T-02-05 | Information Disclosure | `CLERK_SECRET_KEY` exposure | accept | Key only exists in `.env.local` (gitignored); Server Action runs server-side only; no `NEXT_PUBLIC_` prefix used |
</threat_model>
